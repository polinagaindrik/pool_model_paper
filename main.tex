\section{Matrix Notation}

Consider the system of $N$ species in $M$ different states.
The system is described with \ac{ode} in matrix form:
\begin{equation}
    \frac{d}{dt} \tilde X = \tilde A \tilde X
\end{equation}
where tensor $\tilde X$ is of ($NM \times 1$) size:
\begin{equation}
    \tilde X = \begin{pmatrix}
        X_1    \\
        \vdots \\
        X_N
        \end{pmatrix} 
        =   
            \begin{pmatrix}
            \begin{pmatrix}
                x_{11}  \\
                \vdots  \\
                x_{1M}  \\
            \end{pmatrix} \\
                \vdots  \\
            \begin{pmatrix}
                x_{N1}  \\
                \vdots  \\
                x_{NM}
            \end{pmatrix} 
            \end{pmatrix} 
\end{equation}
And tensor  $\tilde A$ is of the size ($NM \times NM$):
\begin{equation}
    \tilde A = \begin{pmatrix}
        A_1^1  & \dots & A_N^1 \\
        \vdots &       & \vdots \\
        A_1^N  & \dots & A_N^N
    \end{pmatrix}
\end{equation}
Where $A_j^i$ is ($M \times M$)-tensor that describes the infuence of species $j$ on species $i$.
\begin{equation}
    A_j^i = \begin{pmatrix}
        a_{j1}^{i1} & \dots & a_{jM}^{i1} \\
        \vdots      &       & \vdots      \\
        a_{j1}^{iM} & \dots & a_{jM}^{iM} \\
    \end{pmatrix}
\end{equation}

Or another way write it. If $\tilde X$ is the $N \times M$ tensor.
\begin{equation}
    \frac{d}{dt} (X_1, ..., X_N) = (X_1, ..., X_N) \begin{pmatrix}
                                                        A_1^1  & \dots & A_1^N \\
                                                        \vdots &       & \vdots \\
                                                        A_N^1  & \dots & A_N^N
                                                    \end{pmatrix}
\end{equation}

\section{Representation with tensor product}
\begin{equation}
    \tilde A \tilde X = \sum_{j=1}^N  \sum_{k=1}^M  \bigotimes_{i=1}^N \mathbf{e}_i \bigotimes_{l=1}^M \mathbf{e}_l a^{il}_{jk} x^{jk} 
    =  \sum_{j=1}^N  \sum_{k=1}^M  x^{jk} \bigotimes_{i=1}^N \mathbf{e}_i \bigotimes_{l=1}^M \mathbf{e}_l a^{il}_{jk}
\end{equation}

\begin{equation}
    \tilde A \tilde X = \bigotimes_{l=1}^N \mathbf{e}_l \bigotimes_{i=1}^M \mathbf{e}_i A^{il} \tilde X 
\end{equation}
where 
\begin{equation}
    A^{il} =  \begin{pmatrix}
        (a^{il}_{11} \dots a^{il}_{1M}) & \dots & (a^{il}_{N1} \dots a^{il}_{NM})
    \end{pmatrix}
\end{equation}
